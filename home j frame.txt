    public ArrayList<Hotel>ListHotels(String ValToSearch)
    {
        ArrayList<Hotel>hotelslist = new ArrayList<>();
        Statement st;
        ResultSet rs;
         String searchQuery = "SELECT place_name,hotel_name,star_rating FROM hotel,place WHERE place_id=place_no ";

        try{
            Connection con = MyConnection.getConnection();
            st = con.createStatement();
                       rs=st.executeQuery(searchQuery);
            Hotel hotel;
            while(rs.next())
            {
                hotel=new Hotel(rs.getString("place_name"),rs.getString("hotel_name"),rs.getInt("star_rating") );              
              hotelslist.add(hotel);
            }
            
        }catch(Exception ex){
            System.out.println(ex.getMessage());
            }
        return hotelslist;
    }
    public void findHotels()
    {
        ArrayList<Hotel>hotel=ListHotels(jText_search.getText());
        DefaultTableModel model= (DefaultTableModel)jTable_hotel.getModel();
        model.setColumnIdentifiers(new Object[]{"place_name","hotel_name","star_ratings"});
        Object[] row = new Object[3];
        for(int i=0;i<hotel.size();i++)
        {
            row[0]=hotel.get(i).getplace_name();
            row[1]=hotel.get(i).gethotel_name();
            row[2]=hotel.get(i).getstar();
            model.addRow(row);
        }
        jTable_hotel.setModel(model);
    }
